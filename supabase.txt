-- Login Token Doğrulama RPC Fonksiyonu (SECURITY DEFINER ile RLS aşarak çalışır)
CREATE OR REPLACE FUNCTION public.validate_login_token(p_token uuid)
RETURNS json
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
    v_driver RECORD;
    v_student RECORD;
BEGIN
    SELECT id, company_id, full_name, phone INTO v_driver
    FROM public.drivers
    WHERE login_token = p_token AND status = 'active'
    LIMIT 1;

    IF FOUND THEN
        RETURN json_build_object(
            'success', true,
            'type', 'driver',
            'user', json_build_object(
                'id', v_driver.id,
                'company_id', v_driver.company_id,
                'full_name', v_driver.full_name,
                'phone', v_driver.phone
            )
        );
    END IF;

    SELECT id, company_id, parent_name, parent_phone, full_name as student_name INTO v_student
    FROM public.students
    WHERE login_token = p_token AND status = 'active'
    LIMIT 1;

    IF FOUND THEN
        RETURN json_build_object(
            'success', true,
            'type', 'parent',
            'user', json_build_object(
                'id', v_student.id,
                'company_id', v_student.company_id,
                'parent_name', v_student.parent_name,
                'parent_phone', v_student.parent_phone,
                'student_name', v_student.student_name
            )
        );
    END IF;

    RETURN json_build_object(
        'success', false,
        'message', 'Geçersiz veya süresi dolmuş giriş kodu.'
    );
END;
$$;

AUTHENTICATION AND REGISTRATION RLS SETUP QUERY:

-- ==========================================
-- AUTHENTICATION & REGISTRATION SQL SETUP
-- ==========================================

-- 1. ROW LEVEL SECURITY (RLS) - HELPER FUNCTION
-- CRITICAL FIX: Changed from LANGUAGE sql to LANGUAGE plpgsql. 
-- SQL functions are often inlined by PostgreSQL, which strips the SECURITY DEFINER context 
-- and executes the query with the caller's privileges, causing infinite RLS recursion.
-- plpgsql functions are never inlined, ensuring the RLS bypass works correctly.
CREATE OR REPLACE FUNCTION public.get_auth_company_id()
RETURNS UUID AS $$
DECLARE
    v_company_id UUID;
BEGIN
    SELECT company_id INTO v_company_id FROM public.users WHERE id = auth.uid() LIMIT 1;
    RETURN v_company_id;
END;
$$ LANGUAGE plpgsql STABLE SECURITY DEFINER SET search_path = public;


-- 2. CREATE RLS POLICIES FOR USERS & COMPANIES

-- NUCLEAR RESET: Clean up all possible old policies
DO $$ 
DECLARE 
    pol record;
BEGIN
    FOR pol IN (SELECT policyname, tablename FROM pg_policies WHERE schemaname = 'public' AND tablename IN ('users', 'companies')) 
    LOOP
        EXECUTE format('DROP POLICY IF EXISTS %I ON public.%I', pol.policyname, pol.tablename);
    END LOOP;
END $$;

-- Companies Table: Essential read/write access
CREATE POLICY "companies_select" ON public.companies FOR SELECT TO authenticated USING (
    owner_id = auth.uid() OR id = public.get_auth_company_id()
);
CREATE POLICY "companies_insert" ON public.companies FOR INSERT TO authenticated WITH CHECK (auth.uid() = owner_id);
CREATE POLICY "companies_update" ON public.companies FOR UPDATE TO authenticated USING (owner_id = auth.uid());

-- Users Table: Decoupled to prevent recursion
CREATE POLICY "users_select_self" ON public.users FOR SELECT TO authenticated USING (id = auth.uid());
CREATE POLICY "users_update_self" ON public.users FOR UPDATE TO authenticated USING (id = auth.uid());
CREATE POLICY "users_delete_self" ON public.users FOR DELETE TO authenticated USING (id = auth.uid());


-- 3. ATOMIC REGISTRATION FUNCTION (RPC)
-- Using SECURITY DEFINER to bypass RLS during initial setup
DROP FUNCTION IF EXISTS public.register_company(UUID, TEXT, TEXT, TEXT);
CREATE OR REPLACE FUNCTION public.register_company(
    p_user_id UUID,
    p_company_name TEXT,
    p_full_name TEXT,
    p_subscription_tier TEXT DEFAULT 'free'
) RETURNS jsonb AS $$
DECLARE
    v_company_id UUID;
    v_user_record RECORD;
BEGIN
    -- 1. Create Company
    INSERT INTO public.companies (company_name, subscription_tier, owner_id)
    VALUES (p_company_name, p_subscription_tier, p_user_id)
    RETURNING id INTO v_company_id;

    -- 2. Create User Profile
    INSERT INTO public.users (id, company_id, full_name, role)
    VALUES (p_user_id, v_company_id, p_full_name, 'owner')
    RETURNING * INTO v_user_record;
    
    RETURN row_to_json(v_user_record)::jsonb;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER SET search_path = public;


-- 4. GRANTS FOR REGISTRATION
GRANT EXECUTE ON FUNCTION public.register_company TO anon, authenticated;
GRANT SELECT, INSERT ON public.companies TO authenticated;
GRANT SELECT, INSERT ON public.users TO authenticated;
